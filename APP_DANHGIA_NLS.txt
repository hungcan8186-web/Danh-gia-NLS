        // --- Khởi tạo Vue Application ---
        const { createApp, ref, reactive, computed, onMounted, onUnmounted } = Vue;

        createApp({
            setup() {
                const role = ref('GV');
                
                // Dữ liệu người dùng
                const userInfo = reactive({
                    name: '',
                    school: '',
                    year: '',
                    department: '',
                    position: '',
                    evaluator: '',
                    date: new Date().toISOString().split('T')[0]
                });

                // Dữ liệu phần đánh giá chung
                const evaluation = reactive({
                    comments: '',
                    suggestions: '',
                    development: '',
                    training: ''
                });

                // Deep clone data to add reactivity properties (score, evidence)
                const addReactiveProps = (dataObj) => {
                    return JSON.parse(JSON.stringify(dataObj)).map(domain => {
                        domain.items = domain.items.map(item => ({ ...item, score: null, evidence: '' }));
                        return domain;
                    });
                };

                const state = reactive({
                    gv: addReactiveProps(gvData),
                    cbql: addReactiveProps(cbqlData)
                });

                const currentCriteria = computed(() => {
                    return role.value === 'GV' ? state.gv : state.cbql;
                });

                // --- Logic tính toán ---
                const totalItems = computed(() => {
                    return currentCriteria.value.reduce((total, domain) => total + domain.items.length, 0);
                });

                const maxScore = computed(() => {
                    return totalItems.value * 3;
                });

                const stats = computed(() => {
                    let total = 0;
                    let completed = 0;
                    currentCriteria.value.forEach(domain => {
                        domain.items.forEach(item => {
                            if (item.score) {
                                total += item.score;
                                completed++;
                            }
                        });
                    });
                    return { total, completed };
                });

                const totalScore = computed(() => stats.value.total);
                const completedItems = computed(() => stats.value.completed);

                // Tính toán tỷ lệ % luôn chính xác dù là GV (20 tiêu chí = max 60đ) hay CBQL (24 tiêu chí = max 72đ)
                const percentage = computed(() => {
                    if (maxScore.value === 0) return 0;
                    return Math.round((totalScore.value / maxScore.value) * 100);
                });

                // Sử dụng chuẩn Tỷ lệ % của Thông tư để xếp loại (Không bị ảnh hưởng bởi số lượng câu)
                const classification = computed(() => {
                    const p = percentage.value;
                    if (p === 0) return { label: 'Chưa đánh giá', textColor: 'text-slate-400', barColor: 'bg-slate-300' };
                    if (p >= 90) return { label: 'Xuất sắc', textColor: 'text-green-600', barColor: 'bg-green-500' };
                    if (p >= 75) return { label: 'Giỏi', textColor: 'text-blue-600', barColor: 'bg-blue-500' };
                    if (p >= 50) return { label: 'Khá', textColor: 'text-yellow-500', barColor: 'bg-yellow-400' };
                    if (p >= 33) return { label: 'Trung bình', textColor: 'text-orange-500', barColor: 'bg-orange-400' };
                    return { label: 'Chưa đạt', textColor: 'text-red-600', barColor: 'bg-red-500' };
                });

                // --- Hành động ---
                const setRole = (newRole) => {
                    if (role.value !== newRole) {
                        if (confirm(`Bạn muốn chuyển sang đánh giá cho ${newRole === 'GV' ? 'Giáo viên' : 'CBQL'}? Dữ liệu hiện tại sẽ được lưu lại tự động.`)) {
                            role.value = newRole;
                        }
                    }
                };

                const selectScore = (item, score) => {
                    item.score = score;
                    saveToLocal();
                };

                const resetForm = () => {
                    if (confirm('Bạn có chắc chắn muốn xóa toàn bộ kết quả đánh giá của vai trò này không?')) {
                        state[role.value.toLowerCase()] = addReactiveProps(role.value === 'GV' ? gvData : cbqlData);
                        evaluation.comments = '';
                        evaluation.suggestions = '';
                        evaluation.development = '';
                        evaluation.training = '';
                        saveToLocal();
                    }
                };

                const printReport = () => {
                    window.print();
                };

                // Auto-save logic
                const saveToLocal = () => {
                    const dataToSave = {
                        role: role.value,
                        userInfo,
                        evaluation,
                        gv: state.gv,
                        cbql: state.cbql
                    };
                    localStorage.setItem('eduDigitalRubricData', JSON.stringify(dataToSave));
                };

                // Scroll interaction for sticky footer
                const isScrollingDown = ref(false);
                let lastScrollPosition = 0;

                const handleScroll = () => {
                    const currentScroll = window.pageYOffset || document.documentElement.scrollTop;
                    if (currentScroll > lastScrollPosition && currentScroll > 100) {
                        isScrollingDown.value = true;
                    } else {
                        isScrollingDown.value = false;
                    }
                    lastScrollPosition = currentScroll <= 0 ? 0 : currentScroll;
                };

                onMounted(() => {
                    const savedData = localStorage.getItem('eduDigitalRubricData');
                    if (savedData) {
                        try {
                            const parsed = JSON.parse(savedData);
                            role.value = parsed.role || 'GV';
                            if (parsed.userInfo) Object.assign(userInfo, parsed.userInfo);
                            if (parsed.evaluation) Object.assign(evaluation, parsed.evaluation);
                            if (parsed.gv) state.gv = parsed.gv;
                            if (parsed.cbql) state.cbql = parsed.cbql;
                        } catch (e) {
                            console.error("Could not load saved data", e);
                        }
                    }
                    window.addEventListener('scroll', handleScroll, { passive: true });
                });

                onUnmounted(() => {
                    window.removeEventListener('scroll', handleScroll);
                });

                // Watchers to auto-save evidence and evaluations
                Vue.watch(() => state.gv, saveToLocal, { deep: true });
                Vue.watch(() => state.cbql, saveToLocal, { deep: true });
                Vue.watch(userInfo, saveToLocal, { deep: true });
                Vue.watch(evaluation, saveToLocal, { deep: true });

                return {
                    role,
                    userInfo,
                    evaluation,
                    currentCriteria,
                    totalScore,
                    maxScore,
                    percentage,
                    completedItems,
                    totalItems,
                    classification,
                    isScrollingDown,
                    setRole,
                    selectScore,
                    resetForm,
                    printReport
                };
            }
        }).mount('#app');
    </script>
